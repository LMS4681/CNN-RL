```python
def render_candidate_mask(
    self, ws: Workspace, candidate: CandidatePlacement
) -> np.ndarray:
    mask = np.zeros((1, self.grid_size, self.grid_size), dtype=np.float32)
    if candidate.position is None:
        return mask
    cx, cy = candidate.position
    self._render_rectangle(
        mask[0], ws, self._compute_scale(ws),
        cx, cy, candidate.length, candidate.breadth, value=1.0,
    )
    return mask
```

Extract the pixel-bound calculation currently embedded in `_render_block()` into

```python
def _render_rectangle(
    self,
    channel: np.ndarray,
    ws: Workspace,
    scale_info: dict,
    center_x: float,
    center_y: float,
    length: float,
    breadth: float,
    value: float,
) -> None:
    scale = scale_info["scale"]
    offset_x = scale_info["offset_x"]
    offset_y = scale_info["offset_y"]
    x0 = center_x - length / 2.0 - ws.origin_x
    y0 = center_y - breadth / 2.0 - ws.origin_y
    px0 = max(0, int(x0 * scale) + offset_x)
    py0 = max(0, int(y0 * scale) + offset_y)
    px1 = min(self.grid_size, int((x0 + length) * scale) + offset_x)
    py1 = min(self.grid_size, int((y0 + breadth) * scale) + offset_y)
    if px1 > px0 and py1 > py0:
        channel[py0:py1, px0:px1] = value
```

```python
def select_workspaces(
    workspaces: List[Workspace],
    active_codes: Optional[List[str]],
) -> List[Workspace]:
    if not active_codes:
        return list(workspaces)
    wanted = {code.strip().upper() for code in active_codes if code.strip()}
    known = {ws.code.upper() for ws in workspaces}
    unknown = sorted(wanted - known)
    if unknown:
        raise ValueError("Unknown active workspace code(s): " + ", ".join(unknown))
    selected = [ws for ws in workspaces if ws.code.upper() in wanted]
    if not selected:
        raise ValueError("At least one active workspace is required.")
    return selected
```

```python
class OrderedBlockEncoder(nn.Module):
    def __init__(self, observation_space: gym.spaces.Dict):
        super().__init__()
        block_dim = observation_space["block"].shape[0]
        self.current = nn.Sequential(
            nn.Linear(block_dim, 64), nn.ReLU(),
            nn.Linear(64, 32), nn.ReLU(),
        )
        self.k = (
            observation_space["future_blocks"].shape[0]
            if "future_blocks" in observation_space.spaces else 0
        )
        self.future = (
            nn.Sequential(
                nn.Linear(self.k * 9, 128), nn.ReLU(),
                nn.Linear(128, 64), nn.ReLU(),
            ) if self.k else None
        )

    def forward(self, observations: dict[str, torch.Tensor]) -> torch.Tensor:
        current = self.current(observations["block"])
        if self.future is None:
            future = current.new_zeros((current.shape[0], 64))
        else:
            mask = observations["future_mask"]
            masked = observations["future_blocks"] * mask.unsqueeze(-1)
            future_input = torch.cat(
                [masked.flatten(1), mask.flatten(1)], dim=1
            )
            future = self.future(future_input)
        return torch.cat([current, future], dim=1)
```

- [ ] **Step 3: Implement a hook-based diagnostic tracker**

```python
class CnnDiagnosticTracker:
    def __init__(self, extractor):
        self.extractor = extractor
        self.module = getattr(extractor, "image_encoder", None)
        self._hooks = []
        self._grad_sq = 0.0
        self._snapshot = None

    def attach(self) -> None:
        if self.module is None:
            return
        self._snapshot = [p.detach().clone() for p in self.module.parameters()]
        self._hooks = [
            p.register_hook(self._capture_gradient)
            for p in self.module.parameters() if p.requires_grad
        ]

    def _capture_gradient(self, grad):
        self._grad_sq += float(grad.detach().square().sum())

    def record_update(self) -> dict[str, float]:
        if self.module is None:
            return {}
        current = [p.detach() for p in self.module.parameters()]
        delta_sq = sum(
            float((now - old).square().sum())
            for now, old in zip(current, self._snapshot)
        )
        metrics = {
            "cnn_gradient_norm": self._grad_sq ** 0.5,
            "cnn_weight_change": delta_sq ** 0.5,
        }
        self._snapshot = [p.clone() for p in current]
        self._grad_sq = 0.0
        return metrics

    def measure_features(
        self, observations: dict[str, torch.Tensor]
    ) -> dict[str, float]:
        if self.module is None:
            return {}
        grids = observations["grids"]
        without_candidate = grids.clone()
        without_candidate[:, :, 3] = 0.0
        with torch.no_grad():
            normal = self.extractor.encode_grids(grids)
            baseline = self.extractor.encode_grids(without_candidate)
        sensitivity = torch.linalg.vector_norm(
            normal - baseline, dim=-1
        ).mean()
        return {
            "workspace_feature_variance": float(
                normal.var(unbiased=False)
            ),
            "candidate_channel_sensitivity": float(sensitivity),
        }

    def close(self) -> None:
        for hook in self._hooks:
            hook.remove()
        self._hooks = []
```

```python
def future_workspace_choice_count(self) -> int:
    if self._placement_simulator is None:
        return 0
    total = 0
    for block_index in self._placement_simulator.upcoming_block_indices(
        self._n_future_blocks
    ):
        block = self._blocks[block_index]
        for workspace in self._workspaces:
            if not all(
                constraint.is_feasible(block, workspace)
                for constraint in self._constraints
            ):
                continue
            trial = block.clone()
            position = workspace.determine_placement_position(
                trial, self._env_date
            )
            if position is None:
                trial.turn()
                position = workspace.determine_placement_position(
                    trial, self._env_date
                )
            total += int(position is not None)
    return total
```
